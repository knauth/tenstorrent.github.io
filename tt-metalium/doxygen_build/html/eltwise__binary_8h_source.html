30ALWI void binary_op_init_common(uint32_t icb0, uint32_t icb1, uint32_t ocb=16)
31{
32 UNPACK(( llk_setup_operands() ));
33 UNPACK(( llk_unpack_AB_hw_configure_disaggregated<DST_ACCUM_MODE>(icb0, icb1) ));
34 UNPACK(( llk_unpack_AB_init<BroadcastType::NONE>(icb0, icb1) ));
35
36 MATH(( llk_math_pack_sync_init<DST_ACCUM_MODE>() ));
37
38 PACK(( llk_pack_hw_configure_disaggregated<false, DST_ACCUM_MODE>(ocb) ));
39 PACK(( llk_pack_init(ocb) ));
40 PACK(( llk_setup_outputs() ));
41 PACK(( llk_pack_dest_init<false, DST_ACCUM_MODE>() ));
42}

174template<bool full_init = false, EltwiseBinaryType eltwise_binary_op_type = ELWADD>
175ALWI void binary_op_specific_init() // TODO(AP): better naming
176{
177 if constexpr (full_init) {
178 if constexpr (eltwise_binary_op_type == ELWADD) {
179 add_tiles_init();
180 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
181 sub_tiles_init();
182 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
183 mul_tiles_init();
184 }
185 } else {
186 if constexpr (eltwise_binary_op_type == ELWADD) {
187 add_tiles_init_nof();
188 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
189 sub_tiles_init_nof();
190 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
191 mul_tiles_init_f();
192 }
193 }
194}
